-- | ensure I can print hashmaps
instance (Pretty k, Pretty v) => Pretty (HashMap k v) where
  pretty = vcat . fmap (\(k, v) -> indent 2 $ fillSep $ punctuate colon [pretty k, indent 4 $ pretty v]) . toList
